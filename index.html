from sklearn.neighbors import KNeighborsClassifier
from sklearn.cluster import KMeans

class RecommendationEngine:
    def __init__(self, n_clusters=5):
        self.kmeans = KMeans(n_clusters)
        self.knn = KNeighborsClassifier(n_neighbors=7)
    
    def fit(self, user_features, exhibitions):
        # Group users by demographics
        clusters = self.kmeans.fit_predict(user_features)
        # Train personalized model
        self.knn.fit(user_features, clusters)
        return self
    
    def recommend(self, new_user):
        cluster = self.knn.predict([new_user])
        return self._score(cluster[0])
